pipeline {
    agent any

    stages {
        stage('Clean Workspace') {
            steps {
                deleteDir()
            }
        }

        stage('Checkout') {
            steps {
                git(
                    url: 'https://github.com/GaurShankarGope/My-First-Jenkin-project.git',
                    branch: 'main',
                    credentialsId: 'github-credentials' // remove if not needed
                )
            }
        }

        stage('Install Dependencies') {
            steps {
                retry(3) {
                    bat 'npm install'
                }
            }
        }

        stage('Build Angular Project') {
            steps {
                bat 'npm run build --prod'
            }
        }

        stage('Archive Build') {
            steps {
                archiveArtifacts artifacts: 'dist/**', followSymlinks: false
            }
        }

        stage('Deploy to IIS') {
            steps {
                // Make sure IIS site folder exists and Jenkins runs as admin
                bat '''
                if not exist "C:\\inetpub\\wwwroot\\MyApp" mkdir "C:\\inetpub\\wwwroot\\MyApp"
                xcopy /E /Y /I dist\\ngrx-project\\* "C:\\inetpub\\wwwroot\\MyApp\\"
                '''
            }
        }

        stage('Restart IIS') {
            steps {
                bat 'iisreset'
            }
        }
    }

    post {
        failure {
            echo 'Build or deployment failed! Check console logs.'
        }
        success {
            echo 'Angular app successfully deployed to IIS!'
        }
    }
}
