pipeline {
    agent any

    stages {
        stage('Clean Workspace') {
            steps {
                deleteDir()
            }
        }

        stage('Checkout') {
            steps {
                git(
                    url: 'https://github.com/GaurShankarGope/My-First-Jenkin-project.git',
                    branch: 'main',
                    credentialsId: 'github-credentials'
                )
            }
        }

        stage('Install Dependencies') {
            steps {
                retry(3) {
                    bat 'npm install'
                }
            }
        }

        stage('Build Angular Project') {
            steps {
                bat 'npm run build --prod'
            }
        }

        stage('Archive Build') {
            steps {
                archiveArtifacts artifacts: 'dist/**', followSymlinks: false
            }
        }

        stage('Deploy to IIS') {
            steps {
                bat '''
                xcopy /E /Y /I dist\\My-First-Jenkin-project\\* "C:\\inetpub\\wwwroot\\MyApp\\"
                '''
            }
        }

        stage('Restart IIS') {
            steps {
                bat 'iisreset'
            }
        }
    }
}
