pipeline {
    agent any

    environment {
        NODE_CMD = "C:\\Program Files\\nodejs\\npm.cmd"
        PROJECT_DIR = "${env.WORKSPACE}\\ngrx-project"
        DIST_DIR = "${env.WORKSPACE}\\dist\\ngrx-project"
        IIS_DIR = "C:\\inetpub\\wwwroot\\ngrx-project"
    }

    stages {
        stage('Checkout SCM') {
            steps {
                checkout scm
            }
        }

        stage('Install Dependencies') {
            steps {
                dir("${PROJECT_DIR}") {
                    // Install Node packages
                    bat "\"${NODE_CMD}\" install"
                }
            }
        }

        stage('Build Angular Project') {
            steps {
                dir("${PROJECT_DIR}") {
                    // Build Angular in production mode
                    bat "\"${NODE_CMD}\" run build -- --configuration production"
                }
            }
        }

        stage('Deploy to IIS') {
            steps {
                // Create IIS directory if it does not exist
                bat "if not exist \"${IIS_DIR}\" mkdir \"${IIS_DIR}\""
                // Copy Angular build files to IIS directory
                bat "xcopy /E /Y /I /C \"${DIST_DIR}\\*\" \"${IIS_DIR}\\\""
            }
        }

        stage('Restart IIS') {
            steps {
                // Restart IIS (use forward slashes to avoid escape issues)
                bat 'C:/Windows/SysNative/iisreset.exe'
                // Alternatively, use:
                // bat 'cmd /c "C:\\Windows\\SysNative\\iisreset.exe"'
            }
        }
    }

    post {
        success {
            echo "Build & deployment succeeded!"
        }
        failure {
            echo "Build or deployment failed! Check console logs."
        }
    }
}
