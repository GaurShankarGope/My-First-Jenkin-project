pipeline {
    agent any

    stages {
        stage('Checkout SCM') {
            steps {
                checkout scm
            }
        }

        stage('Clean Workspace') {
            steps {
                deleteDir()
            }
        }

        stage('Install Dependencies') {
            steps {
                dir('ngrx-project') {  // Run commands inside ngrx-project
                    retry(2) {
                        bat 'npm install'
                    }
                }
            }
        }

        stage('Build Angular Project') {
            steps {
                dir('ngrx-project') {
                    bat 'npm run build'
                }
            }
        }

        stage('Archive Build') {
            steps {
                dir('ngrx-project') {
                    archiveArtifacts artifacts: 'dist/**', fingerprint: true
                }
            }
        }

        stage('Deploy to IIS') {
            steps {
                bat 'if not exist "C:\\inetpub\\wwwroot\\ngrx-project" mkdir "C:\\inetpub\\wwwroot\\ngrx-project"'
                bat 'xcopy /E /Y /I ngrx-project\\dist\\* "C:\\inetpub\\wwwroot\\ngrx-project\\"'
            }
        }

        stage('Restart IIS') {
            steps {
                bat 'C:\\Windows\\System32\\iisreset.exe'
            }
        }
    }

    post {
        failure {
            echo 'Build or deployment failed! Check console logs.'
        }
    }
}
