pipeline {
    agent any

    environment {
        BUILD_DIR = "dist"
        DEPLOY_DIR = "C:\\inetpub\\wwwroot\\ngrx-project"
    }

    stages {

        stage('Checkout SCM') {
            steps {
                // Checkout the repo
                checkout scm
            }
        }

        stage('Clean Workspace') {
            steps {
                deleteDir()
            }
        }

        stage('Install Dependencies') {
            steps {
                retry(2) {
                    // Run npm install in the root where package.json exists
                    bat 'npm install'
                }
            }
        }

        stage('Build Angular Project') {
            steps {
                // Build the Angular project for production
                bat 'npm run build -- --configuration production'
            }
        }

        stage('Archive Build') {
            steps {
                // Archive all files in dist folder
                archiveArtifacts artifacts: "${BUILD_DIR}/**", fingerprint: true
            }
        }

        stage('Deploy to IIS') {
            steps {
                // Create IIS folder if it doesn't exist
                bat "if not exist \"${DEPLOY_DIR}\" mkdir \"${DEPLOY_DIR}\""

                // Copy all built files to IIS folder
                bat "xcopy /E /Y /I ${BUILD_DIR}\\* \"${DEPLOY_DIR}\\\""
            }
        }

        stage('Restart IIS') {
            steps {
                // Restart IIS to apply changes
                bat 'iisreset'
            }
        }
    }

    post {
        failure {
            echo 'Build or deployment failed! Check console logs.'
        }
    }
}
