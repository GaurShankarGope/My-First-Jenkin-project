pipeline {
    agent any

    environment {
        NODE_HOME = "C:\\Program Files\\nodejs"
        PATH = "${env.NODE_HOME};${env.PATH}"
        WORKSPACE_DIR = "${WORKSPACE}"
        DIST_DIR = "${WORKSPACE}\\dist\\ngrx-project"
        IIS_DIR = "C:\\inetpub\\wwwroot\\ngrx-project"
    }

    stages {
        stage('Checkout SCM') {
            steps {
                checkout scm
            }
        }

        stage('Clean Workspace') {
            steps {
                deleteDir()
            }
        }

        stage('Install Dependencies') {
            steps {
                dir("${WORKSPACE_DIR}") {
                    bat '"C:\\Program Files\\nodejs\\npm.cmd" install'
                }
            }
        }

        stage('Build Angular Project') {
            steps {
                dir("${WORKSPACE_DIR}") {
                    bat '"C:\\Program Files\\nodejs\\npm.cmd" run build -- --configuration production'
                }
            }
        }

        stage('Deploy to IIS') {
            steps {
                dir("${WORKSPACE_DIR}") {
                    // Ensure target folder exists
                    bat "if not exist \"${IIS_DIR}\" mkdir \"${IIS_DIR}\""

                    // Copy build files to IIS folder
                    bat "xcopy /E /Y /I \"${DIST_DIR}\\*\" \"${IIS_DIR}\\\""

                    // Restart IIS using PowerShell
                    bat 'powershell -Command "Restart-Service W3SVC -Force"'
                }
            }
        }
    }

    post {
        success {
            echo "Build and Deployment completed successfully!"
        }
        failure {
            echo "Build or deployment failed! Check console logs."
        }
    }
}
