pipeline {
    agent any

    environment {
        WORKSPACE_DIR = "C:\\ProgramData\\Jenkins\\.jenkins\\workspace\\My First Jenkins Application"
        IIS_DIR = "C:\\inetpub\\wwwroot\\ngrx-project"
    }

    stages {

        stage('Clean Workspace') {
            steps {
                // Optional: clean before checkout
                deleteDir()
            }
        }

        stage('Checkout SCM') {
            steps {
                checkout scm
            }
        }

        stage('Install Dependencies') {
            steps {
                dir("${WORKSPACE_DIR}") {
                    bat '"C:\\Program Files\\nodejs\\npm.cmd" install'
                }
            }
        }

        stage('Build Angular Project') {
            steps {
                dir("${WORKSPACE_DIR}") {
                    bat '"C:\\Program Files\\nodejs\\npm.cmd" run build -- --configuration production'
                }
            }
        }

        stage('Deploy to IIS') {
            steps {
                bat "if not exist \"${IIS_DIR}\" mkdir \"${IIS_DIR}\""
                bat "xcopy /E /Y /I \"${WORKSPACE_DIR}\\dist\\*\" \"${IIS_DIR}\\\""
            }
        }

        stage('Restart IIS') {
            steps {
                bat 'C:\\Windows\\Sysnative\\iisreset.exe'
            }
        }
    }

    post {
        success { echo "Build and Deployment Succeeded!" }
        failure { echo "Build or deployment failed! Check console logs." }
    }
}
