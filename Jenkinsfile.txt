pipeline {
    agent any

    stages {
        stage('Checkout SCM') {
            steps {
                checkout scm
            }
        }

        stage('Clean Workspace') {
            steps {
                deleteDir()
            }
        }

        stage('Install Dependencies') {
            steps {
                retry(2) {
                    bat 'npm install'
                }
            }
        }

        stage('Build Angular Project') {
            steps {
                bat 'npm run build --prod'
            }
        }

        stage('Archive Build') {
            steps {
                archiveArtifacts artifacts: 'dist/ngrx-project/**', fingerprint: true
            }
        }

        stage('Deploy to IIS') {
            steps {
                // Ensure target folder exists
                bat 'if not exist "C:\\inetpub\\wwwroot\\MyApp" mkdir "C:\\inetpub\\wwwroot\\MyApp"'

                // Copy files to IIS
                bat 'xcopy /E /Y /I dist\\ngrx-project\\* "C:\\inetpub\\wwwroot\\MyApp\\"'
            }
        }

        stage('Restart IIS') {
            steps {
                // Full path to iisreset
                bat 'C:\\Windows\\System32\\iisreset.exe'
            }
        }
    }

    post {
        failure {
            echo 'Build or deployment failed! Check console logs.'
        }
    }
}
